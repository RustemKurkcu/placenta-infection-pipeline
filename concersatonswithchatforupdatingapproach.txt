run_harmony_compat <- function(seu_obj, batch_col, dims = 1:30) {
  fn <- get("RunHarmony", envir = asNamespace("harmony"))
  fn_formals <- names(formals(fn))
  
  args <- list(
    object = seu_obj,
    group.by.vars = batch_col,
    verbose = FALSE
  )
  
  # Handle API differences between harmony versions.
  if ("reduction" %in% fn_formals) {
    args$reduction <- "pca"
  } else if ("reduction.use" %in% fn_formals) {
    args$reduction.use <- "pca"
  }
  
  if ("dims.use" %in% fn_formals) args$dims.use <- dims
  if ("assay.use" %in% fn_formals) args$assay.use <- "RNA"
  
  do.call(fn, args)
}

downsample_for_speed <- function(seu_obj, max_cells = 120000, seed = 42) {
  n <- ncol(seu_obj)
  if (n <= max_cells) return(seu_obj)
  
  set.seed(seed)
  md <- seu_obj@meta.data
  key <- if (all(c(CFG$cols$cell_type, CFG$cols$donor_id) %in% colnames(md))) {
    paste(md[[CFG$cols$cell_type]], md[[CFG$cols$donor_id]], sep = "__")
  } else {
    rep("all", n)
  }
  
  split_idx <- split(seq_len(n), key)
  per_group <- max(1L, floor(max_cells / length(split_idx)))
  keep <- unlist(lapply(split_idx, function(ix) sample(ix, size = min(length(ix), per_group))), use.names = FALSE)
  
  if (length(keep) > max_cells) keep <- sample(keep, size = max_cells)
  subset(seu_obj, cells = colnames(seu_obj)[sort(keep)])
}

safe_dimplot <- function(seu_obj, reduction, group_by, title_txt) {
  if (!group_by %in% colnames(seu_obj@meta.data)) return(NULL)
  DimPlot(seu_obj, reduction = reduction, group.by = group_by, raster = TRUE) +
    ggtitle(title_txt)
}

save_if_not_null <- function(path, p, w = 12, h = 8) {
  if (!is.null(p)) save_plot(path, p, w = w, h = h)
}

run_harmony_compat <- function(seu_obj, batch_col, dims = 1:30) {
  fn <- get("RunHarmony", envir = asNamespace("harmony"))
  fn_formals <- names(formals(fn))

  args <- list(
    object = seu_obj,
    group.by.vars = batch_col,
    verbose = FALSE
  )

  if ("reduction" %in% fn_formals) {
    args$reduction <- "pca"
  } else if ("reduction.use" %in% fn_formals) {
    args$reduction.use <- "pca"
  }

  if ("dims.use" %in% fn_formals) args$dims.use <- dims
  if ("assay.use" %in% fn_formals) args$assay.use <- "RNA"

  do.call(fn, args)
}

# proportional stratified sampling that better approaches max_cells
downsample_for_speed <- function(seu_obj, max_cells = 120000, seed = 42) {
  n <- ncol(seu_obj)
  if (is.infinite(max_cells) || n <= max_cells) return(seu_obj)

  set.seed(seed)
  md <- seu_obj@meta.data

  key <- if (all(c(CFG$cols$cell_type, CFG$cols$donor_id) %in% colnames(md))) {
    paste(md[[CFG$cols$cell_type]], md[[CFG$cols$donor_id]], sep = "__")
  } else {
    rep("all", n)
  }

  idx_by_group <- split(seq_len(n), key)
  group_sizes <- vapply(idx_by_group, length, integer(1))

  # proportional allocation
  target <- round(group_sizes / sum(group_sizes) * max_cells)
  target[target == 0] <- 1L
  target <- pmin(target, group_sizes)

  # adjust to exact max_cells if needed
  diff_n <- sum(target) - max_cells
  if (diff_n > 0) {
    reducible <- which(target > 1)
    while (diff_n > 0 && length(reducible) > 0) {
      j <- sample(reducible, 1)
      target[j] <- target[j] - 1L
      diff_n <- diff_n - 1L
      reducible <- which(target > 1)
    }
  } else if (diff_n < 0) {
    expandable <- which(target < group_sizes)
    while (diff_n < 0 && length(expandable) > 0) {
      j <- sample(expandable, 1)
      target[j] <- target[j] + 1L
      diff_n <- diff_n + 1L
      expandable <- which(target < group_sizes)
    }
  }

  keep <- unlist(
    Map(function(ix, k) sample(ix, size = k), idx_by_group, target),
    use.names = FALSE
  )

  subset(seu_obj, cells = colnames(seu_obj)[sort(keep)])
}

safe_dimplot <- function(seu_obj, reduction, group_by, title_txt) {
  if (!reduction %in% names(seu_obj@reductions)) return(NULL)

  DimPlot(seu_obj, reduction = reduction, group.by = group_by, raster = TRUE) +
    ggtitle(title_txt)
}

save_if_not_null <- function(path, p, w = 12, h = 8) {

}

prep_for_pca <- function(obj, nfeatures = 3000, npcs = 50, log_file = NULL) {
  DefaultAssay(obj) <- "RNA"

  # Normalize only if needed
  obj <- NormalizeData(obj, verbose = FALSE)

  obj <- tryCatch(
    FindVariableFeatures(obj, nfeatures = nfeatures, verbose = FALSE),
    error = function(e) {
      log_msg("FindVariableFeatures failed at nfeatures=", nfeatures,
              ": ", conditionMessage(e), ". Retrying with 2000.",
              log_file = log_file)
      gc()
      FindVariableFeatures(obj, nfeatures = 2000, verbose = FALSE)
    }
  )

  obj <- ScaleData(obj, features = VariableFeatures(obj), verbose = FALSE)
  obj <- RunPCA(obj, features = VariableFeatures(obj), npcs = npcs, verbose = FALSE)

  obj
}

run_harmony_compat <- function(seu_obj, batch_col, dims = 1:30, reduction_save = "harmony") {
  fn <- get("RunHarmony", envir = asNamespace("harmony"))
  fn_formals <- names(formals(fn))

  args <- list(
    object = seu_obj,
    group.by.vars = batch_col,
    verbose = FALSE
  )

  if ("reduction" %in% fn_formals) {
    args$reduction <- "pca"
  } else if ("reduction.use" %in% fn_formals) {
    args$reduction.use <- "pca"
  }

  if ("dims.use" %in% fn_formals) args$dims.use <- dims
  if ("assay.use" %in% fn_formals) args$assay.use <- "RNA"
  if ("reduction.save" %in% fn_formals) args$reduction.save <- reduction_save

  do.call(fn, args)
}

safe_dimplot <- function(seu_obj, reduction, group_by, title_txt) {
  if (!reduction %in% names(seu_obj@reductions)) return(NULL)
  if (!group_by %in% colnames(seu_obj@meta.data)) return(NULL)

  DimPlot(seu_obj, reduction = reduction, group.by = group_by, raster = TRUE) +
    ggtitle(title_txt)
}

downsample_stratified <- function(seu_obj, max_cells = 30000, seed = 42) {
  n <- ncol(seu_obj)
  if (n <= max_cells) return(seu_obj)

  set.seed(seed)
  md <- seu_obj@meta.data

  key <- if (all(c(CFG$cols$cell_type, CFG$cols$donor_id) %in% colnames(md))) {
    paste(md[[CFG$cols$cell_type]], md[[CFG$cols$donor_id]], sep = "__")
  } else {
    rep("all", n)
  }

  idx_by_group <- split(seq_len(n), key)
  group_sizes <- vapply(idx_by_group, length, integer(1))

  target <- round(group_sizes / sum(group_sizes) * max_cells)
  target[target == 0] <- 1L
  target <- pmin(target, group_sizes)

  diff_n <- sum(target) - max_cells
  if (diff_n > 0) {
    reducible <- which(target > 1)
    while (diff_n > 0 && length(reducible) > 0) {
      j <- sample(reducible, 1)
      target[j] <- target[j] - 1L
      diff_n <- diff_n - 1L
      reducible <- which(target > 1)
    }
  } else if (diff_n < 0) {
    expandable <- which(target < group_sizes)
    while (diff_n < 0 && length(expandable) > 0) {
      j <- sample(expandable, 1)
      target[j] <- target[j] + 1L
      diff_n <- diff_n + 1L
      expandable <- which(target < group_sizes)
    }
  }

  keep <- unlist(
    Map(function(ix, k) sample(ix, size = k), idx_by_group, target),
    use.names = FALSE
  )

  subset(seu_obj, cells = colnames(seu_obj)[sort(keep)])
}

neighbor_purity <- function(emb, labels, k = 15) {
  # emb: matrix cells x dims
  # labels: vector
  if (nrow(emb) < (k + 1)) return(NA_real_)

  d <- as.matrix(dist(emb))
  diag(d) <- Inf
  nn <- apply(d, 1, function(x) order(x)[1:k])

  if (is.vector(nn)) nn <- matrix(nn, nrow = 1)

  pur <- sapply(seq_len(nrow(emb)), function(i) {
    mean(labels[nn[, i]] == labels[i], na.rm = TRUE)
  })
  mean(pur, na.rm = TRUE)
}

mixing_score <- function(emb, batch, k = 15) {
  # higher = more donor mixing
  if (nrow(emb) < (k + 1)) return(NA_real_)

  d <- as.matrix(dist(emb))
  diag(d) <- Inf
  nn <- apply(d, 1, function(x) order(x)[1:k])

  if (is.vector(nn)) nn <- matrix(nn, nrow = 1)

  mix <- sapply(seq_len(nrow(emb)), function(i) {
    mean(batch[nn[, i]] != batch[i], na.rm = TRUE)
  })
  mean(mix, na.rm = TRUE)
}

condition_neighbor_enrichment <- function(emb, cond, k = 15) {
  # higher = local condition coherence
  if (length(unique(cond)) < 2 || nrow(emb) < (k + 1)) return(NA_real_)

  d <- as.matrix(dist(emb))
  diag(d) <- Inf
  nn <- apply(d, 1, function(x) order(x)[1:k])

  if (is.vector(nn)) nn <- matrix(nn, nrow = 1)

  score <- sapply(seq_len(nrow(emb)), function(i) {
    mean(cond[nn[, i]] == cond[i], na.rm = TRUE)
  })
  mean(score, na.rm = TRUE)
}

prep_pca <- function(obj, nfeatures, npcs = 60) {
  DefaultAssay(obj) <- "RNA"
  obj <- NormalizeData(obj, verbose = FALSE)
  obj <- FindVariableFeatures(obj, nfeatures = nfeatures, verbose = FALSE)
  obj <- ScaleData(obj, features = VariableFeatures(obj), verbose = FALSE)
  obj <- RunPCA(obj, features = VariableFeatures(obj), npcs = npcs, verbose = FALSE)
  obj
}

pc_var_table <- function(seu_obj) {
  stdev <- seu_obj[["pca"]]@stdev
  data.frame(
    PC = seq_along(stdev),
    stdev = stdev,
    var_explained = (stdev^2) / sum(stdev^2),
    cumvar = cumsum((stdev^2) / sum(stdev^2))
  )
}

> downsample_stratified <- function(seu_obj, max_cells = 30000, seed = 42) {
+   n <- ncol(seu_obj)
+   if (n <= max_cells) return(seu_obj)
+   
+   set.seed(seed)
+   md <- seu_obj@meta.data
+   
+   key <- if (all(c(CFG$cols$cell_type, CFG$cols$donor_id) %in% colnames(md))) {
+     paste(md[[CFG$cols$cell_type]], md[[CFG$cols$donor_id]], sep = "__")
+   } else {
+     rep("all", n)
+   }
+   
+   idx_by_group <- split(seq_len(n), key)
+   group_sizes <- vapply(idx_by_group, length, integer(1))
+   
+   target <- round(group_sizes / sum(group_sizes) * max_cells)
+   target[target == 0] <- 1L
+   target <- pmin(target, group_sizes)
+   
+   diff_n <- sum(target) - max_cells
+   if (diff_n > 0) {
+     reducible <- which(target > 1)
+     while (diff_n > 0 && length(reducible) > 0) {
+       j <- sample(reducible, 1)
+       target[j] <- target[j] - 1L
+       diff_n <- diff_n - 1L
+       reducible <- which(target > 1)
+     }
+   } else if (diff_n < 0) {
+     expandable <- which(target < group_sizes)
+     while (diff_n < 0 && length(expandable) > 0) {
+       j <- sample(expandable, 1)
+       target[j] <- target[j] + 1L
+       diff_n <- diff_n + 1L
+       expandable <- which(target < group_sizes)
+     }
+   }
+   
+   keep <- unlist(
+     Map(function(ix, k) sample(ix, size = k), idx_by_group, target),
+     use.names = FALSE
+   )
+   
+   subset(seu_obj, cells = colnames(seu_obj)[sort(keep)])
+ }

> neighbor_purity <- function(emb, labels, k = 15) {
+   # emb: matrix cells x dims
+   # labels: vector

+   
+   d <- as.matrix(dist(emb))

Yep — and the good news is the new error is much more straightforward than the first one.

What broke this time

The key failure is:

Error in `object[[reduction]]`:
! ‘harmony_4000_40’ not found in this Seurat object

That means Harmony ran, but it did not save the reduction under the custom name harmony_4000_40.

Why this happened:

your wrapper checks formals() on RunHarmony

but with S3/generic dispatch, that is not a reliable way to detect whether reduction.save is supported

so Harmony likely saved the reduction as the default "harmony" instead of "harmony_4000_40"

That matches Harmony’s documented Seurat method, where reduction.save is the argument that controls the slot name.

Fastest fix

Because each loop starts with:

tmp <- obj

you do not need unique Harmony reduction names at all.

Just use the default "harmony" every time.

Replace this whole block:

if (requireNamespace("harmony", quietly = TRUE)) {
  harm_name <- paste0("harmony_", n_hvg, "_", n_pc)

  tmp <- run_harmony_compat(
    seu_obj = tmp,
    batch_col = batch_col,
    dims = 1:n_pc,
    reduction_save = harm_name
  )

  tmp <- RunUMAP(
    tmp,
    dims = 1:n_pc,
    reduction = harm_name,
    reduction.name = paste0("umap_harmony_", n_hvg, "_", n_pc),
    n.neighbors = 30,
    min.dist = 0.3,
    verbose = FALSE
  )
}

with this:

if (requireNamespace("harmony", quietly = TRUE)) {
  tmp <- harmony::RunHarmony(

Second problem: std::bad_alloc at 2000 HVGs

This part:

Error: std::bad_alloc
Called from: SparseRowVar2(...)

is happening during FindVariableFeatures(), not because 2000 is biologically wrong, but because the loop is keeping too much around in memory while repeatedly creating large Seurat objects. FindVariableFeatures() identifies variable genes by modeling mean–variance behavior, and ElbowPlot() is the lighter-weight way to judge PCs, whereas JackStraw() is explicitly the heavier formal test.

So yes: you absolutely can process one setting, save outputs, then remove objects and run garbage collection. That is fine practice here. gc() is literally R’s garbage collection mechanism, and using rm() plus gc() inside big loops is a normal way to control memory pressure in large workflows.

run_harmony_compat <- function(seu_obj, batch_col, dims = 1:30, reduction_save = "harmony") {
  fn <- get("RunHarmony", envir = asNamespace("harmony"))
  fn_formals <- names(formals(fn))
  
  args <- list(
    object = seu_obj,
    group.by.vars = batch_col,
    verbose = FALSE
  )
  
  if ("reduction" %in% fn_formals) {
    args$reduction <- "pca"
  } else if ("reduction.use" %in% fn_formals) {
    args$reduction.use <- "pca"
  }
  
  if ("dims.use" %in% fn_formals) args$dims.use <- dims
  if ("assay.use" %in% fn_formals) args$assay.use <- "RNA"
  if ("reduction.save" %in% fn_formals) args$reduction.save <- reduction_save
  
  do.call(fn, args)
}

downsample_stratified <- function(seu_obj, max_cells = 30000, seed = 42) {
  n <- ncol(seu_obj)
  if (n <= max_cells) return(seu_obj)
  
  set.seed(seed)
  md <- seu_obj@meta.data
  
  key <- if (all(c(CFG$cols$cell_type, CFG$cols$donor_id) %in% colnames(md))) {
    paste(md[[CFG$cols$cell_type]], md[[CFG$cols$donor_id]], sep = "__")
  } else {
    rep("all", n)
  }
  
  idx_by_group <- split(seq_len(n), key)
  group_sizes <- vapply(idx_by_group, length, integer(1))
  
  target <- round(group_sizes / sum(group_sizes) * max_cells)
  target[target == 0] <- 1L
  target <- pmin(target, group_sizes)
  
  diff_n <- sum(target) - max_cells
  if (diff_n > 0) {
    reducible <- which(target > 1)
    while (diff_n > 0 && length(reducible) > 0) {
      j <- sample(reducible, 1)
      target[j] <- target[j] - 1L
      diff_n <- diff_n - 1L
      reducible <- which(target > 1)
    }
  } else if (diff_n < 0) {
    expandable <- which(target < group_sizes)
    while (diff_n < 0 && length(expandable) > 0) {
      j <- sample(expandable, 1)
      target[j] <- target[j] + 1L
      diff_n <- diff_n + 1L
      expandable <- which(target < group_sizes)
    }
  }
  
  keep <- unlist(
    Map(function(ix, k) sample(ix, size = k), idx_by_group, target),
    use.names = FALSE
  )
  
  subset(seu_obj, cells = colnames(seu_obj)[sort(keep)])
}

neighbor_purity <- function(emb, labels, k = 15) {
  # emb: matrix cells x dims
  # labels: vector
  if (nrow(emb) < (k + 1)) return(NA_real_)
  
  d <- as.matrix(dist(emb))
  diag(d) <- Inf
  nn <- apply(d, 1, function(x) order(x)[1:k])
  
  if (is.vector(nn)) nn <- matrix(nn, nrow = 1)
  
  pur <- sapply(seq_len(nrow(emb)), function(i) {
    mean(labels[nn[, i]] == labels[i], na.rm = TRUE)
  })
  mean(pur, na.rm = TRUE)
}

mixing_score <- function(emb, batch, k = 15) {
  # higher = more donor mixing
  if (nrow(emb) < (k + 1)) return(NA_real_)
  
  d <- as.matrix(dist(emb))
  diag(d) <- Inf
  nn <- apply(d, 1, function(x) order(x)[1:k])
  
  if (is.vector(nn)) nn <- matrix(nn, nrow = 1)
  
  mix <- sapply(seq_len(nrow(emb)), function(i) {
    mean(batch[nn[, i]] != batch[i], na.rm = TRUE)
  })
  mean(mix, na.rm = TRUE)
}

condition_neighbor_enrichment <- function(emb, cond, k = 15) {
  # higher = local condition coherence
  if (length(unique(cond)) < 2 || nrow(emb) < (k + 1)) return(NA_real_)
  
  d <- as.matrix(dist(emb))
  diag(d) <- Inf
  nn <- apply(d, 1, function(x) order(x)[1:k])
  
  if (is.vector(nn)) nn <- matrix(nn, nrow = 1)
  
  score <- sapply(seq_len(nrow(emb)), function(i) {
    mean(cond[nn[, i]] == cond[i], na.rm = TRUE)
  })
  mean(score, na.rm = TRUE)
}

  obj <- ScaleData(obj, features = VariableFeatures(obj), verbose = FALSE)
  obj <- RunPCA(obj, features = VariableFeatures(obj), npcs = npcs, verbose = FALSE)
  obj
}

pc_var_table <- function(seu_obj) {
  stdev <- seu_obj[["pca"]]@stdev
  data.frame(
    PC = seq_along(stdev),
    stdev = stdev,
    var_explained = (stdev^2) / sum(stdev^2),
    cumvar = cumsum((stdev^2) / sum(stdev^2))
  )
}

Seurat documents FindVariableFeatures() with default selection.method = "vst" and default nfeatures = 2000, but that is a default rather than a rule; the function is meant to identify genes that are unusually variable relative to their mean expression. Seurat also notes that ElbowPlot() is a fast way to find a reasonable PC cutoff and is much faster than JackStraw, which is why deferring JackStraw here is a good call. Harmony’s Seurat method expects PCA already computed and supports dims.use plus reduction.save, so using the standard "harmony" reduction name is clean and robust. DietSeurat() is also explicitly intended to slim objects by removing reductions/graphs you do not need.

# =========================================================
# 07b_parameter_sensitivity.R
#
# PURPOSE
# -------
# Test a small, biologically reasonable grid of:
#   - HVGs: 3000, 4000, 5000
#   - PCs : 30, 40, 50
#
# Why this exists:
#   1) Placenta is highly heterogeneous, so 2000 HVGs may be too conservative.
#   2) We want an evidence-based choice of dimensionality for manuscript-quality work.
#   3) We want to preserve biologically meaningful structure:
#        - major placental cell types
#        - infection/time patterns
#        - donor correction without over-correction
#        - FIB2 / stromal / trophoblast / immune structure
#
# DESIGN CHOICES
# --------------
# - Keep ElbowPlot diagnostics
# - Skip JackStraw for now (computationally expensive)

safe_dimplot <- function(seu_obj, reduction, group_by, title_txt) {
  if (!reduction %in% names(seu_obj@reductions)) return(NULL)
  if (!group_by %in% colnames(seu_obj@meta.data)) return(NULL)

  DimPlot(seu_obj, reduction = reduction, group.by = group_by, raster = TRUE) +
    ggtitle(title_txt)
}

save_if_not_null <- function(path, p, w = 10, h = 8) {
  if (!is.null(p)) save_plot(path, p, w = w, h = h)
}

# Simple elbow suggestion:
# returns the first PC after which marginal gain becomes small
# This is a heuristic, not a substitute for biological review.
suggest_elbow_pc <- function(seu_obj, min_pc = 10, max_pc = 50, rel_drop_thresh = 0.10) {
  stdev <- seu_obj[["pca"]]@stdev
  max_pc <- min(max_pc, length(stdev))

  if (max_pc < (min_pc + 2)) return(NA_integer_)

  vars <- stdev[1:max_pc]^2
  deltas <- -diff(vars)  # variance drop from PC_i to PC_{i+1}

  # relative drop compared with the first drop
  rel <- deltas / deltas[1]

  # first PC where the incremental drop becomes "small"
  idx <- which(seq_along(rel) >= min_pc & rel < rel_drop_thresh)[1]

  if (is.na(idx)) {
    return(max_pc)
  } else {
    return(idx)
  }
}

prep_pca <- function(obj, nfeatures, npcs = 50) {
  DefaultAssay(obj) <- "RNA"

  # Drop old reductions/graphs to keep the working copy lighter.
  # Seurat provides DietSeurat specifically for slimming objects. 
  obj <- DietSeurat(
    object = obj,
    assays = "RNA",
    layers = c("counts", "data"),
    dimreducs = NULL,
    graphs = NULL,
    misc = TRUE
  )

  obj <- NormalizeData(obj, verbose = FALSE)
  obj <- FindVariableFeatures(
    obj,
    selection.method = "vst",
    nfeatures = nfeatures,
    verbose = FALSE
  )
  obj <- ScaleData(obj, features = VariableFeatures(obj), verbose = FALSE)
  obj <- RunPCA(obj, features = VariableFeatures(obj), npcs = npcs, verbose = FALSE)

  obj
}

pc_var_table <- function(seu_obj) {
  stdev <- seu_obj[["pca"]]@stdev
  data.frame(
    PC = seq_along(stdev),
    stdev = stdev,
    var_explained = (stdev^2) / sum(stdev^2),
    cumvar = cumsum((stdev^2) / sum(stdev^2))
  )
}

run_harmony_safe <- function(seu_obj, batch_col, dims_use) {
  # Harmony documentation indicates PCA must already exist.
  # We use the default reduction.save = "harmony" within each fresh loop object.
  harmony::RunHarmony(
    object = seu_obj,
    group.by.vars = batch_col,
    reduction = "pca",
    dims.use = dims_use,
    assay.use = "RNA",
    reduction.save = "harmony",
    verbose = FALSE
  )
}

    prep_pca(seu, nfeatures = n_hvg, npcs = max(pc_grid)),
    error = function(e) {
      log_msg("prep_pca failed for HVG=", n_hvg, " :: ", conditionMessage(e), log_file = log_file)
      stop(e)
    }
  )

  pca_tbl <- pc_var_table(obj)
  elbow_pc_auto <- suggest_elbow_pc(obj, min_pc = 10, max_pc = max(pc_grid), rel_drop_thresh = 0.10)

  pca_tbl$auto_elbow_pc <- elbow_pc_auto

    row.names = FALSE

  # Save HVG-level object after PCA
  qs::qsave(

  # Elbow plot with automatic suggestion annotated
  p_elbow <- ElbowPlot(obj, ndims = max(pc_grid), reduction = "pca") +
    geom_vline(xintercept = elbow_pc_auto, linetype = "dashed") +
    ggtitle(paste0("ElbowPlot | HVGs=", n_hvg, " | auto elbow ~ PC ", elbow_pc_auto))

  save_plot(
    file.path(CFG$dirs$figures, paste0("FigParam_Elbow_HVG", n_hvg, ".pdf")),
    p_elbow, w = 9, h = 6
  )

  # Variable feature plot
  p_vf <- VariableFeaturePlot(obj) +
    ggtitle(paste0("VariableFeaturePlot | HVGs=", n_hvg))

  save_plot(
    file.path(CFG$dirs$figures, paste0("FigParam_VariableFeatures_HVG", n_hvg, ".pdf")),
    p_vf, w = 9, h = 6
  )

  # Per-PC loop
  for (n_pc in pc_grid) {

    log_msg("Running HVG=", n_hvg, " | PCs=", n_pc, log_file = log_file)

    # Fresh copy from the HVG-specific PCA object
    tmp <- obj

    # -----------------------------
    # Non-integrated UMAP
    # -----------------------------
    tmp <- RunUMAP(
      tmp,
      dims = 1:n_pc,
      reduction = "pca",
      reduction.name = "umap_rna",
      n.neighbors = 30,
      min.dist = 0.3,
      verbose = FALSE
    )

    # -----------------------------
    # Harmony + Harmony UMAP
    # -----------------------------
    if (requireNamespace("harmony", quietly = TRUE)) {
      tmp <- tryCatch(
        run_harmony_safe(tmp, batch_col = batch_col, dims_use = 1:n_pc),
        error = function(e) {
          log_msg("Harmony failed for HVG=", n_hvg, " | PC=", n_pc, " :: ", conditionMessage(e), log_file = log_file)
          tmp
        }
      )

      if ("harmony" %in% names(tmp@reductions)) {
        tmp <- RunUMAP(
          tmp,
          dims = 1:n_pc,
          reduction = "harmony",
          reduction.name = "umap_harmony",
          n.neighbors = 30,
          min.dist = 0.3,
          verbose = FALSE
        )
      }
    }

    # -----------------------------
    # Save object for this combination
    # -----------------------------
    if (isTRUE(save_combo_objects)) {
      qs::qsave(
        x = tmp,
        file = file.path(obj_dir, paste0("seu_HVG", n_hvg, "_PC", n_pc, ".qs"))
      )

    # -----------------------------
    # RNA UMAP panel
    # -----------------------------
    p1 <- safe_dimplot(tmp, "umap_rna", CFG$cols$cell_type,
                       paste0("RNA UMAP | cell type | HVG=", n_hvg, " PC=", n_pc))
    p2 <- safe_dimplot(tmp, "umap_rna", batch_col,

    p3 <- safe_dimplot(tmp, "umap_rna", CFG$cols$infection,
                       paste0("RNA UMAP | infection | HVG=", n_hvg, " PC=", n_pc))
    p4 <- safe_dimplot(tmp, "umap_rna", "condition",
                       paste0("RNA UMAP | condition | HVG=", n_hvg, " PC=", n_pc))

    if (!is.null(p1) && !is.null(p2) && !is.null(p3) && !is.null(p4)) {
      panel <- (p1 | p2) / (p3 | p4)
      save_plot(
        file.path(CFG$dirs$figures, paste0("FigParam_RNA_HVG", n_hvg, "_PC", n_pc, ".pdf")),
        panel, w = 16, h = 12
      )
    }

    # -----------------------------
    # Harmony UMAP panel
    # -----------------------------
    if ("umap_harmony" %in% names(tmp@reductions)) {
      h1 <- safe_dimplot(tmp, "umap_harmony", CFG$cols$cell_type,
                         paste0("Harmony UMAP | cell type | HVG=", n_hvg, " PC=", n_pc))
      h2 <- safe_dimplot(tmp, "umap_harmony", batch_col,
                         paste0("Harmony UMAP | donor | HVG=", n_hvg, " PC=", n_pc))
      h3 <- safe_dimplot(tmp, "umap_harmony", CFG$cols$infection,
                         paste0("Harmony UMAP | infection | HVG=", n_hvg, " PC=", n_pc))
      h4 <- safe_dimplot(tmp, "umap_harmony", "condition",
                         paste0("Harmony UMAP | condition | HVG=", n_hvg, " PC=", n_pc))

      stringsAsFactors = FALSE
    )

    # -----------------------------
    # Clean up large temp objects
    # -----------------------------
    rm(list = intersect(
      c("tmp", "p1", "p2", "p3", "p4", "panel", "h1", "h2", "h3", "h4", "hpanel"),

    ))
    cleanup_big_objects()
  }

  rm(list = intersect(c("obj", "pca_tbl", "p_elbow", "p_vf", "elbow_pc_auto"), ls()))
  cleanup_big_objects()

each HVG-level PCA object

That is good practice here because it makes the sweep restartable and auditable.

Proposed primarily by Christopher Redman in the 1990s, it breaks the disease down into a placental root cause and a maternal systemic reaction.
Here is a breakdown of the two stages:
Stage 1: Abnormal Placentation (The Placenta)
This occurs early in pregnancy (1st trimester).
The Issue: Normally, specialized fetal cells (trophoblasts) remodel the mother’s spiral arteries into wide, low-resistance channels to ensure steady blood flow to the fetus.
The Failure: In PE, this remodeling is "defective." The arteries remain narrow, leading to placental ischemia (restricted blood flow) and oxidative stress. The placenta becomes "stressed" because it isn't getting enough oxygen or nutrients.
Stage 2: Maternal Systemic Syndrome (The Mother)
This occurs later in pregnancy (after 20 weeks) and produces the clinical symptoms.
The Bridge: The stressed placenta releases "factors" (like sFlt-1 and inflammatory cytokines) into the mother’s bloodstream.
The Result: These factors cause widespread endothelial dysfunction (damage to the lining of blood vessels). This leads to the classic signs of preeclampsia:
High blood pressure (hypertension).
Leaky kidneys (proteinuria).
Organ damage (liver, brain, or lungs).
Why the image mentions a "Gap"
While the Two-Stage Model is the gold standard, it has flaws that your image highlights:
The "Subset" Problem: Many women have "defective placentation" (Stage 1) but never develop the full maternal syndrome (Stage 2). The model doesn't explain what "triggers" the transition in some women but not others.
The Inflammatory Trigger: Modern research suggests there might be a "Stage 0" or an external trigger. Your image points to oral microbiome dysbiosis (specifically Fn, or Fusobacterium nucleatum).
The Microbiome Link: The theory is that bacteria from the mouth (due to gum disease/periodontitis) can travel through the blood to the placenta. This infection might be the "extra push" that causes a struggling placenta to release the toxic factors that trigger Stage 2.
In short: The Two-Stage Model explains how the body fails, but it doesn't fully explain why it happens to some people and not others—which is where the study of the microbiome and inflammation comes in.

what is sFlt-1 , whchi cells produce normally which in preclampsia
sFlt-1 (soluble fms-like tyrosine kinase-1) is a protein that plays a critical role in the development of preeclampsia. It is a "soluble" version of a receptor that normally sits on the surface of blood vessels.
What is sFlt-1?
Think of sFlt-1 as a "molecular sponge" or decoy receptor.
The Normal Process: To stay healthy, blood vessels need "growth factors" called VEGF (Vascular Endothelial Growth Factor) and PlGF (Placental Growth Factor). These factors act like "food" for the lining of the blood vessels (the endothelium), keeping them flexible and open.
The Problem in Preeclampsia: sFlt-1 floats freely in the mother’s blood and binds to these growth factors before they can reach the blood vessels. By "soaking up" the growth factors, sFlt-1 leaves the mother’s blood vessels "starved," leading to the inflammation, high blood pressure, and kidney damage seen in preeclampsia.
Which cells produce sFlt-1?
The production of sFlt-1 changes significantly between a healthy state and preeclampsia:
State	Primary Cell Sources	Key Characteristics
Normal (Healthy)	Endothelial cells (lining blood vessels), Monocytes (immune cells), and the Placenta.	Produced in small, regulated amounts to manage blood vessel growth. Levels rise slightly toward the end of a normal pregnancy.
Preeclampsia	Placental Trophoblasts (specifically syncytiotrophoblasts and cytotrophoblasts).	The placenta becomes the "factory" for sFlt-1. Due to low oxygen (hypoxia), these cells overproduce sFlt-1 at levels 20–50 times higher than normal.
The "Placental-Specific" Variant
Recent research has identified a specific version of this protein called sFlt-1 e15a.
In healthy people, the body mostly makes a version called sFlt-1 i13.
In women with preeclampsia, the placenta pumps out massive amounts of the e15a variant. This variant is unique to humans and higher primates, which may explain why preeclampsia is a disease that only affects humans.
Why this matters clinically
Doctors now use the sFlt-1/PlGF ratio as a diagnostic tool.
High Ratio: Indicates too much "sponge" (sFlt-1) and too little "food" (PlGF), suggesting a high risk that preeclampsia will develop or worsen within the next 1–4 weeks.
Low Ratio: Highly reliable for "ruling out" the disease for a short period, helping avoid unnecessary hospitalizations.

What is not yet publication-worthy

The main statistical weakness is that the current inference is still too close to cell-level exploration and small-sample summaries. For publication-level claims, especially with repeated donors and infection/time structure, you should make the inferential unit the donor/sample, not the cell. fit_feature_by_celltype <- function(df, response) {
  out <- lapply(split(df, df$cell_type), function(dd) {
    # require enough samples
    if (nrow(dd) < 6 || length(unique(dd$infection)) < 2) return(NULL)

    dd$donor_id  <- factor(dd$donor_id)
    dd$infection <- factor(dd$infection)
    dd$hpi       <- factor(dd$hpi)

    formula_txt <- paste(response, "~ donor_id + infection * hpi")
    fit <- lm(as.formula(formula_txt), data = dd)
    aov_tab <- anova(fit)

    data.frame(
      cell_type = unique(dd$cell_type),
      term = rownames(aov_tab),
      df = aov_tab$Df,
      F_value = aov_tab$`F value`,
      p_value = aov_tab$`Pr(>F)`
    )
  })

  bind_rows(out)
}

What is weak right now

The biggest weakness is statistical unit choice. For publication, the inferential unit should be the donor/sample, not the cell. Current best-practice guidance is very clear that condition-level DE in scRNA-seq should use pseudobulk aggregation or an explicit random-effect framework to account for within-sample correlation, because naive cell-level testing inflates evidence.

The second weakness is claim resolution. In your current Seurat object, FLT1 is a gene-level signal. That is useful, but it is not the same thing as measuring the placental soluble isoform sFLT1-e15a. The literature supports e15a as a placental, primate-specific, and major pathogenic soluble FLT1 isoform in PE, so your scRNA analysis should be framed as a FLT1/PGF proxy, not a direct e15a readout.

The third weakness is repo organization. The infection repo’s public GitHub tree currently exposes mainly helper files in R/ (config.R, helpers_io.R, helpers_plot.R, helpers_scores.R), while the numbered working scripts appear more fully only in the uploaded bundles; that makes reproduction from GitHub alone weaker than it should be. Meanwhile, the spatial repo root mixes code, PDFs, presentations, outputs, scripts, and data-heavy artifacts in one place, which is workable for a lab notebook but not ideal for long-term reproducibility.

# Comprehensive review of the two repositories and the current analysis process

## Repositories reviewed

### 1) RustemKurkcu/placenta-infection-pipeline
This repository is the focused single-cell infection pipeline and currently exposes helper infrastructure in the `R/` directory (`config.R`, `helpers_io.R`, `helpers_plot.R`, `helpers_scores.R`). The actively used numbered scripts were reviewed from the uploaded script bundle and outputs.

### 2) RustemKurkcu/hPlacenta-architecture
This repository is a fork of `jian-shu-lab/hPlacenta-architecture` and includes `BO'sWork`, `RPAC`, `config`, `data`, `docs`, `output`, `scripts`, plus `RUN_PIPELINE_MINIMAL.R`, `RUN_PIPELINE_ADVANCED.R`, and an optional `06C_cellchat_optional.R` module. The updated README positions it as a “Vicious Cycle” spatial pipeline integrating a multiome RNA reference with Slide-tags and STARmap-ISS, and explicitly adds k-sensitivity, spatial nulls, and pseudobulk niche DE by donor/week.

## What is strong in the infection pipeline

- Clean modular structure.
- Good use of logs, legends, and saved outputs.
- Existing cell-type annotations are reusable and should be preserved.
- Strong biological starting point because the Cell Systems explant study already demonstrated donor-aware differential analysis with limma/edgeR, CellPhoneDB-based interaction analysis, and pathogen-specific responses.

## What is weak in the infection pipeline

- Too much emphasis on cell-level or score-level exploration without enough donor-level inference.
- Early parameter-sweep implementation was not disk-safe.
- Harmony sensitivity was useful as QC, but it is not itself a biological result.
- Current proxy analyses need better calibration, stronger negative controls, and explicit limitations.

## What is strong in the spatial architecture pipeline

- The advanced README is already thinking like a reviewer: k-robustness, week-stratified rewiring, density-aware nulls, and donor/week pseudobulk are exactly the types of safeguards that prevent overclaiming.
- The optional CellChat module is carefully written to skip gracefully and to run by week.
- The project is already aimed at spatial vulnerability and niche logic, which is the right conceptual complement to the infection pipeline.

## What is weak in the spatial architecture pipeline

- It is still presented partly as a broad “vulnerability” framework rather than being tightly yoked to the MegL toxic-switch model.
- The README says some key EA genes are absent from measured spatial panels, which means the organoid or cross-platform layer becomes essential rather than optional.
- There is a risk of using many composite spatial scores without enough orthogonal validation.

## Best combined strategy

- Use the infection pipeline to establish **who changes under infection**.
- Use the architecture pipeline to establish **where the permissive niche exists**.
- Use public PE datasets to establish **whether the same host-side program is clinically relevant**.
- Use a pathogen specificity matrix to establish **why this is not just generic infection biology**.

## Most important methodological upgrades

1. Donor-aware pseudobulk DE and module analysis by cell type.
2. Reference mapping from architecture labels into infection fibroblast/perivascular states.
3. Spatial hotspot projection of EA and FLT1-related host modules.
4. Targeted rather than generic cell-cell communication analysis.
5. Public PE validation.
6. Disk-safe object management and session-level reproducibility.

## Best candidate headline for the thesis/paper

A host-defined placental ethanolamine niche and stromal–vascular signaling program explain how Fusobacterium nucleatum could convert placental colonization into a preeclampsia-like antiangiogenic syndrome.
